CREATE TABLE TB_ALUNO(

NOME_ALUNO VARCHAR2(100));

ALTER TABLE TB_ALUNO ADD CONSTRAINT tb_aluno_pk PRIMARY KEY(id_aluno);

CREATE TABLE TB_CURSO(
ID_CURSO NUMBER NOT NULL,
NOME_CURSO VARCHAR2(100));

ALTER TABLE TB_CURSO ADD CONSTRAINT tb_curso_fk PRIMARY KEY (id_curso);

CREATE TABLE TB_TURMA(
id_aluno NOT NULL,
id_curso NOT NULL,
CONSTRAINT tb_turma_id_aluno_fk FOREIGN KEY (id_aluno) REFERENCES TB_ALUNO(id_aluno),
CONSTRAINT tb_turma_id_curso_fk FOREIGN KEY (id_curso) REFERENCES TB_CURSO (id_curso));

CREATE TABLE TB_CONTATO(
id_aluno NOT NULL,
telefone NUMBER,
email VARCHAR2(50),
CONSTRAINT tb_contato_id_aluno_fk FOREIGN KEY (id_aluno) REFERENCES TB_ALUNO (id_aluno));

CREATE TABLE TB_LOGIN(
id_aluno NOT NULL,
senha VARCHAR2(100) NOT NULL, 
turno VARCHAR(15),
CONSTRAINT tb_login_id_aluno_fk FOREIGN KEY (id_aluno) REFERENCES TB_ALUNO (id_aluno));
