SELECT NOME_PIZZA, CATEGORIA, PRECO_PIZZA FROM TB_PIZZA WHERE PRECO_PIZZA IN (SELECT MAX(PRECO_PIZZA) AS PRECO_PIZZA FROM TB_PIZZA GROUP BY CATEGORIA);

SELECT NOME_PIZZA, COUNT(PC.ID_INGREDIENTE)QUANTIDADE_INGREDIENTE FROM TB_PIZZA P
INNER JOIN TB_PIZZA_COMPOSICAO PC ON P.ID_PIZZA = PC.ID_PIZZA
GROUP BY NOME_PIZZA ORDER BY QUANTIDADE_INGREDIENTE ASC FETCH FIRST 1 ROW ONLY;
