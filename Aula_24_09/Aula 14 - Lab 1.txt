SELECT MIN(SALARIO_ATOR) AS MENOR_SALARIO FROM TB_FILME;

SELECT GENERO_FILME AS GENERO, AVG(SALARIO_ATOR) AS MEDIA_SALARIO FROM TB_FILME GROUP BY GENERO_FILME;

SELECT NOME_FILME, MAX(SALARIO_ATOR) FROM TB_FILME GROUP BY NOME_FILME;

SELECT ATOR, COUNT(*) AS TOTAL_ATUACOES FROM TB_FILME GROUP BY ATOR ORDER BY TOTAL_ATUACOES ASC FETCH FIRST 1 ROW ONLY;

SELECT GENERO_FILME, COUNT(DISTINCT NOME_FILME) AS QUANTIDADE_FILMES FROM TB_FILME GROUP BY GENERO_FILME;

SELECT NOME_FILME, COUNT(*) AS QUANTIDADE_ATOR FROM TB_FILME GROUP BY NOME_FILME ORDER BY QUANTIDADE_ATOR DESC FETCH FIRST 1 ROW ONLY;

SELECT GENERO_FILME, SUM(SALARIO_ATOR) AS SOMA_SALARIO FROM TB_FILME GROUP BY GENERO_FILME ORDER BY SOMA_SALARIO DESC FETCH FIRST 1 ROW ONLY;
